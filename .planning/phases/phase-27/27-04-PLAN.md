          allStartPts[r][c] = { x: p.x, y: p.y };
        }
      }

      // Compute edge vector + outward normal (D-14 trapezoid path; reduces to world axis when rectangular).
      const edge = sideCfg.edgeAt();
      const a0 = allStartPts[edge[0].r][edge[0].c];
      const a1 = allStartPts[edge[1].r][edge[1].c];
      // Edge in screen-pixel space:
      const evx = (a1.x - a0.x) * vw;
      const evy = (a1.y - a0.y) * vh;
      const elen = Math.sqrt(evx * evx + evy * evy);
      // Edge unit vector:
      const eux = elen > 1e-3 ? evx / elen : 1;
      const euy = elen > 1e-3 ? evy / elen : 0;
      // Outward normal in screen pixels (rotate 90° to the side away from the grid center):
      // For TOP, outward is up: normal should point away from the grid centroid.
      // We choose the sign based on the side's outwardDY/outwardDX hint in the side config.
      let nx = -euy;
      let ny =  eux;
      // Flip normal if it points the wrong way relative to outwardDX/outwardDY.
      const outwardSign = (sideCfg.outwardDX !== 0)
        ? Math.sign(sideCfg.outwardDX * nx)
        : Math.sign(sideCfg.outwardDY * ny);
      if (outwardSign < 0) { nx = -nx; ny = -ny; }

      squishDragState = {
        sideCfg,
        sideKey,
        startClientX: e.clientX,
        startClientY: e.clientY,
        allStartPts,
        // Drag delta projects onto outward normal (negative for "compressing toward anchor" — i.e. dragging top bar DOWN).
        nx, ny,
      };

      // Visual feedback via handle-ui (UI-SPEC §B9 dragging state).
      const ui = _getSquishUiHandle();
      if (ui && typeof ui.setSquishBarDragVisual === "function") ui.setSquishBarDragVisual(sideKey, true);

      try { e.currentTarget.setPointerCapture?.(e.pointerId); } catch {}
      document.addEventListener("pointermove", onSquishDragMove);
      document.addEventListener("pointerup", onSquishDragEnd);
      document.addEventListener("pointercancel", onSquishDragEnd);
    }

    function onSquishDragMove(e) {
      if (!squishDragState) return;
      e.preventDefault();
      const vw = window.innerWidth;
      const vh = window.innerHeight;
      const rows = grid.srcYs.length;
      const cols = grid.srcXs.length;
      const sp = squishDragState.allStartPts;
      const { sideCfg, nx, ny } = squishDragState;

      // Drag delta in screen pixels.
      const dxPx = e.clientX - squishDragState.startClientX;
      const dyPx = e.clientY - squishDragState.startClientY;
      // Project onto outward normal — POSITIVE outward means the dragged edge moves outward (expanding the board).
      // NEGATIVE outward means inward (squishing toward the anchor).
      const projOutwardPx = dxPx * nx + dyPx * ny;

      // Convert that pixel-projection back to normalized x/y components on the dragged edge.
      const dxNormOnEdge = (projOutwardPx * nx) / vw;
      const dyNormOnEdge = (projOutwardPx * ny) / vh;

      // 1) Move every point on the DRAGGED edge along (dxNormOnEdge, dyNormOnEdge):
      const edge = sideCfg.edgeAt();
      // The dragged edge consists of all points whose (r,c) matches one of the corners' row OR col,
      // depending on edge orientation:
      //   TOP edge: row 0, all cols.   BOTTOM: lastRow, all cols.
      //   LEFT edge: col 0, all rows.  RIGHT: lastCol, all rows.
      // Determine via edgeAt corners — they share either the row index or the col index.
      const sharedRow = edge[0].r === edge[1].r ? edge[0].r : null;
      const sharedCol = edge[0].c === edge[1].c ? edge[0].c : null;

      if (sharedRow !== null) {
        // Top or bottom edge: row = sharedRow, all cols
        for (let c = 0; c < cols; c++) {
          const start = sp[sharedRow][c];
          setPoint(sharedRow, c, Math.max(0, Math.min(1, start.x + dxNormOnEdge)), Math.max(0, Math.min(1, start.y + dyNormOnEdge)));
        }
      } else if (sharedCol !== null) {
        // Left or right edge: col = sharedCol, all rows
        for (let r = 0; r < rows; r++) {
          const start = sp[r][sharedCol];
          setPoint(r, sharedCol, Math.max(0, Math.min(1, start.x + dxNormOnEdge)), Math.max(0, Math.min(1, start.y + dyNormOnEdge)));
        }
      }

      // 2) Anchor edge (OPPOSITE) stays fixed — no mutation needed (its values match sp[anchorR][anchorC]).
      // 3) Interior proportional scaling — mirrors onLineDragMove isEdgeRow/isEdgeCol blocks (lines 452-492).
      if (sharedRow !== null) {
        const lastRow = grid.srcYs.length - 1;
        // sharedRow is either 0 (top) or lastRow (bottom). The anchor row is the opposite.
        for (let ri = 1; ri < lastRow; ri++) {
          for (let ci = 0; ci < cols; ci++) {
            const oldTop = sp[0][ci].y;
            const oldBottom = sp[lastRow][ci].y;
            const newTop = getPoint(0, ci).y;
            const newBottom = getPoint(lastRow, ci).y;
            const oldRange = oldBottom - oldTop;
            const newRange = newBottom - newTop;
            if (Math.abs(oldRange) > 1e-6) {
              const t = (sp[ri][ci].y - oldTop) / oldRange;
              setPoint(ri, ci, getPoint(ri, ci).x, newTop + t * newRange);
            }
            // Also scale X proportionally for trapezoid edge-perpendicular squish:
            const oldLeft = sp[0][ci].x;
            const oldBottomX = sp[lastRow][ci].x;
            const newLeft = getPoint(0, ci).x;
            const newBottomX = getPoint(lastRow, ci).x;
            const oldRangeX = oldBottomX - oldLeft;
            const newRangeX = newBottomX - newLeft;
            if (Math.abs(oldRangeX) > 1e-6) {
              const tX = (sp[ri][ci].x - oldLeft) / oldRangeX;
              setPoint(ri, ci, newLeft + tX * newRangeX, getPoint(ri, ci).y);
            }
          }
        }
      } else if (sharedCol !== null) {
        const lastCol = grid.srcXs.length - 1;
        for (let ci = 1; ci < lastCol; ci++) {
          for (let ri = 0; ri < rows; ri++) {
            const oldLeft = sp[ri][0].x;
            const oldRight = sp[ri][lastCol].x;
            const newLeft = getPoint(ri, 0).x;
            const newRight = getPoint(ri, lastCol).x;
            const oldRangeX = oldRight - oldLeft;
            const newRangeX = newRight - newLeft;
            if (Math.abs(oldRangeX) > 1e-6) {
              const t = (sp[ri][ci].x - oldLeft) / oldRangeX;
              setPoint(ri, ci, newLeft + t * newRangeX, getPoint(ri, ci).y);
            }
            // Trapezoid: also scale Y proportionally.
            const oldTopY = sp[ri][0].y;
            const oldRightY = sp[ri][lastCol].y;
            const newTopY = getPoint(ri, 0).y;
            const newRightY = getPoint(ri, lastCol).y;
            const oldRangeY = oldRightY - oldTopY;
            const newRangeY = newRightY - newTopY;
            if (Math.abs(oldRangeY) > 1e-6) {
              const tY = (sp[ri][ci].y - oldTopY) / oldRangeY;
              setPoint(ri, ci, getPoint(ri, ci).x, newTopY + tY * newRangeY);
            }
          }
        }
      }

      positionHandles();
      positionRotateHandles();   // also re-positions squish bars via Task 1 hook
      drawLines();
      applyTransform();
      if (typeof ctx.renderRoomOverlay === "function") ctx.renderRoomOverlay();
    }

    function onSquishDragEnd() {
      if (!squishDragState) return;
      const sideKey = squishDragState.sideKey;
      squishDragState = null;
      const ui = _getSquishUiHandle();
      if (ui && typeof ui.setSquishBarDragVisual === "function") ui.setSquishBarDragVisual(sideKey, false);
      document.removeEventListener("pointermove", onSquishDragMove);
      document.removeEventListener("pointerup", onSquishDragEnd);
      document.removeEventListener("pointercancel", onSquishDragEnd);
      saveToLocalStorage();
      try { window.TT_BEAMER_RUNTIME_PROJECTION_PROFILE_PERSISTENCE?.notifyDirtyChanged?.(); } catch {}
    }
