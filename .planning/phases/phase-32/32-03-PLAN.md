**Step 2 — Create `test/phase-32-settings-ui.test.mjs`:**

This file tests the panel module via its global IIFE export. The module loads as a side-effect of import — it sets `window.TT_BEAMER_SETTINGS_SERVER_RENDERING_PANEL`. Use a JSDOM-free approach: build a minimal `mockWindow` with `document.querySelectorAll` returning fake radio elements, then dynamically import the module after assigning `globalThis.window = mockWindow; globalThis.document = mockWindow.document;`.

```javascript
import { test } from "node:test";
import { strict as assert } from "node:assert";

// Helper: build a fake radio input
function makeRadio(value) {
  let listener = null;
  let checked = false;
  return {
    value: String(value),
    get checked() { return checked; },
    set checked(v) { checked = Boolean(v); },
    addEventListener(ev, fn) { if (ev === "change") listener = fn; },
    triggerChange() { checked = true; if (listener) listener({ target: { checked, value } }); },
  };
}

function makeCheckbox() {
  let listener = null;
  let checked = false;
  return {
    get checked() { return checked; },
    set checked(v) { checked = Boolean(v); },
    addEventListener(ev, fn) { if (ev === "change") listener = fn; },
    setChecked(v) {
      checked = Boolean(v);
      if (listener) listener({ target: { checked } });
    },
  };
}

function loadPanelModule() {
  // Module is an IIFE that mutates `window`. Set up minimal globals.
  const panelHolder = {};
  globalThis.window = panelHolder;
  // Re-import via cache-busting query param so each test starts fresh
  const ts = Date.now() + Math.random();
  return import(`../src/app/lib/ui/settings/server-rendering-panel.js?cb=${ts}`).then(() => {
    const mod = panelHolder.TT_BEAMER_SETTINGS_SERVER_RENDERING_PANEL;
    if (!mod || typeof mod.initServerRenderingPanel !== "function") {
      throw new Error("panel module did not expose initServerRenderingPanel");
    }
    return mod;
  });
}

test("reflectConfig sets streamFpsCap radio checked", async () => {
  const radios = [makeRadio(30), makeRadio(45), makeRadio(60), makeRadio(0)];
  const refs = {
    ssrEncoderSelect: { addEventListener() {}, value: "auto" },
    ssrStreamFpsCapRadios: radios,
    ssrAlignModeBoostToggle: makeCheckbox(),
    ssrServerRenderingStatus: { textContent: "" },
  };
  const mod = await loadPanelModule();
  let snapshotCb = null;
  mod.initServerRenderingPanel({
    refs,
    emitLiveMutation: () => {},
    fetchGlobalDefaults: async () => ({ serverRendering: { streamFpsCap: 45, alignModeBoost: true } }),
    onSnapshot: (cb) => { snapshotCb = cb; },
  });
  // fetchGlobalDefaults is async; let microtasks flush
  await new Promise(r => setTimeout(r, 50));
  assert.equal(radios[0].checked, false, "30 should not be checked");
  assert.equal(radios[1].checked, true,  "45 should be checked");
  assert.equal(radios[2].checked, false, "60 should not be checked");
  assert.equal(radios[3].checked, false, "0/native should not be checked");
  assert.equal(refs.ssrAlignModeBoostToggle.checked, true);
});

test("reflectConfig sets streamFpsCap=0 (Native) radio checked", async () => {
```
