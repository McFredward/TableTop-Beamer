```javascript
import { existsSync } from "node:fs";

/**
 * Probe libva runtime availability — independent of ffmpeg.
 * Chromium's libwebrtc VaapiVideoEncoder uses libva directly, so VAAPI
 * encode can be available even when ffmpeg lacks `h264_vaapi`. This probe
 * checks the libva shared library presence in the standard paths.
 */
export function probeLibvaRuntime() {
  const candidates = [
    "/usr/lib/x86_64-linux-gnu/libva.so.2",
    "/usr/lib/aarch64-linux-gnu/libva.so.2",
    "/usr/local/lib/libva.so.2",
  ];
  return candidates.some((p) => existsSync(p));
}
```

3. Backwards compat: existing calls without `opts.probe` still work; defaults call both `probeVaapiDevice()` and `probeLibvaRuntime()`.
4. ENCODER_PRIORITY ordering preserved (`vaapi` inserted between `nvenc` and `videotoolbox`).

```javascript
export async function detectAvailableEncoders(opts = {}) {
  const probe = opts.probe ?? {
    ffmpegEncoders: await probeFfmpegEncoders(),
    hasVaapiDevice: probeVaapiDevice(),
    hasLibva: probeLibvaRuntime(),  // ← NEW
  };
  // Backfill if caller provided partial probe (Wave-0 test convenience)
  if (probe.hasLibva === undefined) probe.hasLibva = probeLibvaRuntime();
  if (probe.hasVaapiDevice === undefined) probe.hasVaapiDevice = probeVaapiDevice();
  if (!Array.isArray(probe.ffmpegEncoders)) probe.ffmpegEncoders = [];

  const result = [];
  if (probe.ffmpegEncoders.includes("h264_nvenc")) result.push("nvenc");
  // VAAPI: trust either ffmpeg OR (chromium-libva pair). Do NOT require ffmpeg's h264_vaapi.
  if (probe.hasVaapiDevice && probe.hasLibva) result.push("vaapi");
  else if (probe.ffmpegEncoders.includes("h264_vaapi") && probe.hasVaapiDevice) result.push("vaapi");
  if (probe.ffmpegEncoders.includes("h264_videotoolbox")) result.push("videotoolbox");
  result.push("x264-software");
  return result;
}
```
