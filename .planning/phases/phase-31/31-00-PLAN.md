    Concrete content:
    ```js
    import { test } from "node:test";
    import assert from "node:assert/strict";
    import { spawn } from "node:child_process";
    import { existsSync, mkdirSync, writeFileSync, readFileSync, unlinkSync } from "node:fs";
    import path from "node:path";

    const ALARM_MP3 = path.resolve("./resources/sounds/alarm.mp3");
    const TMPDIR = "/tmp/ttbeamer-wave0";
    const HTML_FIXTURE = path.join(TMPDIR, "audio-fixture.html");
    const CAPTURED_WEBM = path.join(TMPDIR, "captured.webm");
    const DECODED_WAV = path.join(TMPDIR, "decoded.wav");
    const DISPLAY = ":99";

    function hasBin(name) {
      try {
        const r = spawn("which", [name]);
        return new Promise((res) => r.on("exit", (c) => res(c === 0)));
      } catch { return Promise.resolve(false); }
    }

    test("Wave 0 audio gate: Xvfb + Chromium + puppeteer-stream capture alarm.mp3 with non-silent RMS", { timeout: 90_000 }, async (t) => {
      // Skip-gate: only run if user-explicitly opts in (smoke is heavy + needs Xvfb)
      if (process.env.WAVE0_AUDIO_SMOKE !== "1") {
        t.skip("Set WAVE0_AUDIO_SMOKE=1 to run the audio-capture smoke (requires Xvfb + Chromium + ffmpeg on PATH)");
        return;
      }

      // 1. Preflight: required binaries
      assert.ok(await hasBin("Xvfb"), "Xvfb not on PATH — install via `apt install xvfb`");
      assert.ok(await hasBin("ffmpeg"), "ffmpeg not on PATH");
      assert.ok(existsSync(ALARM_MP3), `alarm.mp3 not found at ${ALARM_MP3}`);

      // 2. Tmp dir
      mkdirSync(TMPDIR, { recursive: true });

      // 3. Build a tiny HTML fixture that auto-plays alarm.mp3 (file:// URL)
      writeFileSync(HTML_FIXTURE, `<!doctype html><html><body>
        <audio id="a" autoplay src="file://${ALARM_MP3}"></audio>
        <script>document.getElementById("a").play().catch(e => document.title = "PLAY_FAIL:" + e.message);</script>
      </body></html>`);

      // 4. Spawn Xvfb
      const xvfb = spawn("Xvfb", [DISPLAY, "-screen", "0", "640x480x24"], { stdio: "ignore" });
      await new Promise(r => setTimeout(r, 500));
      t.after(() => xvfb.kill("SIGTERM"));

      // 5. Launch headful Chromium via puppeteer-stream and capture audio
      const { launch, getStream } = await import("puppeteer-stream");
      const browser = await launch({
        executablePath: process.env.SSR_BROWSER_BIN || "/usr/bin/chromium",
        headless: false, // CRITICAL: NOT --headless=new — disables audio
        defaultViewport: { width: 640, height: 480 },
        args: [
          "--no-sandbox",
          "--autoplay-policy=no-user-gesture-required",
          "--use-gl=egl",
          `--display=${DISPLAY}`,
        ],
        env: { ...process.env, DISPLAY },
      });
      t.after(() => browser.close());

      const page = await browser.newPage();
      await page.goto(`file://${HTML_FIXTURE}`);

      const stream = await getStream(page, { audio: true, video: false, frameSize: 100 });
      const captureFile = (await import("node:fs")).createWriteStream(CAPTURED_WEBM);
      stream.pipe(captureFile);

      // 6. Capture for 4 seconds (alarm.mp3 is ~3s)
      await new Promise(r => setTimeout(r, 4000));
      stream.destroy();
      await new Promise(r => captureFile.on("close", r));

      // 7. Decode webm -> wav via ffmpeg
      await new Promise((res, rej) => {
        const ff = spawn("ffmpeg", ["-y", "-i", CAPTURED_WEBM, "-ac", "1", "-ar", "16000", DECODED_WAV], { stdio: "pipe" });
        ff.on("exit", (c) => c === 0 ? res() : rej(new Error("ffmpeg decode failed")));
      });

      // 8. RMS check: read WAV PCM, compute RMS, assert > 0.01 (silence floor)
      const wav = readFileSync(DECODED_WAV);
      // PCM s16le starts at offset 44 in canonical WAV
      let sumSq = 0; let n = 0;
      for (let i = 44; i + 1 < wav.length; i += 2) {
        const s = wav.readInt16LE(i) / 32768;
        sumSq += s * s; n++;
      }
      const rms = Math.sqrt(sumSq / Math.max(1, n));
      console.log(`[wave0-audio] RMS amplitude: ${rms.toFixed(4)} (threshold > 0.01)`);
      assert.ok(rms > 0.01, `Audio capture appears silent (RMS=${rms}). D-D2 escalation: see RESEARCH.md § Audio-Capture Risk fallbacks.`);

      // cleanup
      try { unlinkSync(HTML_FIXTURE); unlinkSync(CAPTURED_WEBM); unlinkSync(DECODED_WAV); } catch {}
    });

    test("Wave 0 scaffold: alarm.mp3 fixture exists in resources/sounds/", () => {
      assert.ok(existsSync("./resources/sounds/alarm.mp3"), "alarm.mp3 must exist as the audio gate fixture");
    });
    ```

    ```js
    // src/server/server-encoder-detect.mjs
    // Phase 31: hardware-agnostic h264 encoder auto-detection at server boot.
    // The user's project is published; this module replaces hardcoded VAAPI/Intel
    // assumptions with a probe of what is ACTUALLY available on the host.
    //
    // Priority: nvenc > vaapi > videotoolbox > x264-software.
    // x264-software is ALWAYS available as universal fallback on any modern OS.

    import { spawn } from "node:child_process";
    import { existsSync } from "node:fs";
    import { platform } from "node:os";

    export const ENCODER_PRIORITY = ["nvenc", "vaapi", "videotoolbox", "x264-software"];

    /**
     * Probe ffmpeg for available h264 encoder names.
     * Returns array like ["h264_nvenc", "h264_vaapi", "libx264"] or [] on failure.
     */
    async function probeFfmpegEncoders() {
      return new Promise((resolve) => {
        try {
          const proc = spawn("ffmpeg", ["-hide_banner", "-encoders"], { stdio: ["ignore", "pipe", "pipe"] });
          let buf = "";
          proc.stdout.on("data", (d) => { buf += d.toString("utf8"); });
          proc.on("error", () => resolve([]));
          proc.on("exit", () => {
            const found = [];
            if (/\bh264_nvenc\b/.test(buf)) found.push("h264_nvenc");
            if (/\bh264_vaapi\b/.test(buf)) found.push("h264_vaapi");
            if (/\bh264_videotoolbox\b/.test(buf)) found.push("h264_videotoolbox");
            if (/\blibx264\b/.test(buf)) found.push("libx264");
            resolve(found);
          });
        } catch { resolve([]); }
      });
    }

    function probeNvidiaSmi() {
      return new Promise((resolve) => {
        try {
          const proc = spawn("nvidia-smi", ["-L"], { stdio: ["ignore", "pipe", "pipe"] });
          proc.on("error", () => resolve(false));
          proc.on("exit", (code) => resolve(code === 0));
        } catch { resolve(false); }
      });
    }

    function probeVaapiDevice() {
      // /dev/dri/renderD128 (and similar) implies VAAPI-capable iGPU/dGPU under Linux.
      return existsSync("/dev/dri/renderD128") || existsSync("/dev/dri/renderD129");
    }

    /**
     * Detect available h264 encoders. Returns priority-ordered list.
     * x264-software is ALWAYS appended as universal fallback.
     *
     * @param {object} [opts]
     * @param {() => Promise<{ffmpegEncoders:string[], hasVaapiDevice:boolean, hasNvidiaSmi:boolean, platform:string}>} [opts.probe] mock injector for tests
     * @returns {Promise<string[]>}
     */
    export async function detectAvailableEncoders(opts = {}) {
      const probe = opts.probe ?? (async () => ({
        ffmpegEncoders: await probeFfmpegEncoders(),
        hasVaapiDevice: probeVaapiDevice(),
        hasNvidiaSmi: await probeNvidiaSmi(),
        platform: platform(),
      }));
      const env = await probe();
      const enc = env.ffmpegEncoders ?? [];
      const result = [];
      // NVENC: ffmpeg lists h264_nvenc AND nvidia-smi succeeds (driver loaded)
      if (enc.includes("h264_nvenc") && env.hasNvidiaSmi) result.push("nvenc");
      // VAAPI: ffmpeg lists h264_vaapi AND a /dev/dri/renderD12x exists
      if (enc.includes("h264_vaapi") && env.hasVaapiDevice) result.push("vaapi");
      // VideoToolbox: ffmpeg lists h264_videotoolbox (typically darwin)
      if (enc.includes("h264_videotoolbox")) result.push("videotoolbox");
      // x264-software: ALWAYS present as universal fallback
      result.push("x264-software");
      return result;
    }

    /**
     * Pick the highest-priority encoder from an availability list.
     * @param {string[]} available
     * @returns {string}
     */
    export function pickPreferredEncoder(available) {
      if (!Array.isArray(available) || available.length === 0) {
        throw new Error("pickPreferredEncoder: empty availability list (detection bug — x264-software must always be present)");
      }
      for (const candidate of ENCODER_PRIORITY) {
        if (available.includes(candidate)) return candidate;
      }
      // Should be unreachable if detectAvailableEncoders is used (it always appends x264-software)
      return available[0];
    }
    ```
