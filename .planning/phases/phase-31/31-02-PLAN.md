    ```js
    import * as mediasoup from "mediasoup";

    const RTC_MIN_PORT = Number(process.env.SSR_RTC_MIN_PORT ?? 40000);
    const RTC_MAX_PORT = Number(process.env.SSR_RTC_MAX_PORT ?? 40100);
    const ANNOUNCED_IP = process.env.SSR_ANNOUNCED_IP ?? null; // null = let mediasoup pick
    const LISTEN_IP = process.env.SSR_LISTEN_IP ?? "0.0.0.0";

    const MEDIA_CODECS = [
      { kind: "audio", mimeType: "audio/opus", clockRate: 48000, channels: 2 },
      {
        kind: "video",
        mimeType: "video/H264",
        clockRate: 90000,
        parameters: {
          "packetization-mode": 1,
          "profile-level-id": "42e01f",
          "level-asymmetry-allowed": 1,
        },
      },
    ];

    let activeWorker = null;
    let activeRouter = null;

    export async function bootMediasoupRouter({ logger = console } = {}) {
      if (activeRouter) return { worker: activeWorker, router: activeRouter };
      activeWorker = await mediasoup.createWorker({
        logLevel: "warn",
        rtcMinPort: RTC_MIN_PORT,
        rtcMaxPort: RTC_MAX_PORT,
      });
      activeWorker.on("died", (err) => {
        logger.error(`[ssr-mediasoup] worker died: ${err?.message ?? "unknown"}`);
        activeWorker = null;
        activeRouter = null;
      });
      activeRouter = await activeWorker.createRouter({ mediaCodecs: MEDIA_CODECS });
      logger.info(`[ssr-mediasoup] router up — codecs: H264 + Opus, ports ${RTC_MIN_PORT}-${RTC_MAX_PORT}`);
      return { worker: activeWorker, router: activeRouter };
    }

    export async function createWebRtcTransport({ router } = {}) {
      const r = router ?? activeRouter;
      if (!r) throw new Error("createWebRtcTransport: router not initialized");
      const transport = await r.createWebRtcTransport({
        listenIps: [{ ip: LISTEN_IP, announcedIp: ANNOUNCED_IP }],
        enableUdp: true,
        enableTcp: true,
        preferUdp: true,
        // Bitrate target derived from the active Stream-Quality preset (Plan 01 resolveEncoderConfig).
        // Default 8 Mbit/s for `balanced` (NVENC/VAAPI), 4 Mbit/s for `low-latency` (x264-software fallback).
        initialAvailableOutgoingBitrate: Number(process.env.SSR_INITIAL_BITRATE ?? 8_000_000),
      });
      return transport;
    }

    export function getRouter() { return activeRouter; }

    export async function shutdownMediasoupRouter() {
      if (activeRouter) { try { activeRouter.close(); } catch {} activeRouter = null; }
      if (activeWorker) { try { activeWorker.close(); } catch {} activeWorker = null; }
    }
    ```

    ```js
    import { createHash } from "node:crypto";
    import { createWebRtcTransport, getRouter } from "./ssr-mediasoup-router.mjs";

    const KNOWN_ACTIONS = new Set([
      "get-router-rtp-capabilities",
      "create-send-transport",
      "create-recv-transport",
      "connect-transport",
      "produce",
      "consume",
      "resume-consumer",
    ]);

    // Per-connection state
    function makeConnState() {
      return {
        sendTransport: null,
        recvTransport: null,
        producers: new Map(), // id -> producer
        consumers: new Map(), // id -> consumer
        role: null, // "ssr-tab" | "consumer" | null
      };
    }

    // V5 ASVS: validate dtlsParameters shape
    function validateDtlsParameters(params) {
      if (!params || typeof params !== "object") return false;
      if (!Array.isArray(params.fingerprints) || params.fingerprints.length === 0) return false;
      for (const fp of params.fingerprints) {
        if (typeof fp.algorithm !== "string" || typeof fp.value !== "string") return false;
      }
      if (params.role && !["client", "server", "auto"].includes(params.role)) return false;
      return true;
    }

    // V5 ASVS: validate rtpParameters minimal shape
    function validateRtpParameters(rtp) {
      return rtp && typeof rtp === "object"
        && Array.isArray(rtp.codecs) && rtp.codecs.length > 0
        && Array.isArray(rtp.encodings);
    }

    export function attachWebRtcSignaling(server, { logger = console } = {}) {
      // GLOBAL state of producers (Producer is created when SSR tab calls "produce";
      // each new consumer connects and calls "consume" against the LATEST producer).
      const state = {
        videoProducer: null,
        audioProducer: null,
      };

      server.on("upgrade", (req, socket) => {
        const url = new URL(req.url || "/", "http://localhost");
        if (url.pathname !== "/api/webrtc/signal") return;

        const role = url.searchParams.get("role") || "consumer";
        // Spoofing guard (RESEARCH § Security Domain): role=ssr-tab restricted to localhost
        if (role === "ssr-tab" && req.socket.remoteAddress !== "127.0.0.1" && req.socket.remoteAddress !== "::1" && req.socket.remoteAddress !== "::ffff:127.0.0.1") {
          logger.warn(`[ssr-signal] rejecting non-localhost ssr-tab from ${req.socket.remoteAddress}`);
          socket.destroy();
          return;
        }

        const key = req.headers["sec-websocket-key"];
        if (!key) { socket.destroy(); return; }
        const accept = createHash("sha1").update(`${key}258EAFA5-E914-47DA-95CA-C5AB0DC85B11`).digest("base64");
        socket.write([
          "HTTP/1.1 101 Switching Protocols",
          "Upgrade: websocket",
          "Connection: Upgrade",
          `Sec-WebSocket-Accept: ${accept}`,
          "", "",
        ].join("\r\n"));

        const conn = makeConnState();
        conn.role = role;
        logger.info(`[ssr-signal] connect role=${role} addr=${req.socket.remoteAddress}`);

        function send(obj) {
          const json = JSON.stringify(obj);
          const len = Buffer.byteLength(json);
          let header;
          if (len < 126) header = Buffer.from([0x81, len]);
          else if (len < 65536) header = Buffer.from([0x81, 126, (len >> 8) & 0xff, len & 0xff]);
          else { const b = Buffer.alloc(10); b[0] = 0x81; b[1] = 127; b.writeBigUInt64BE(BigInt(len), 2); header = b; }
          socket.write(Buffer.concat([header, Buffer.from(json)]));
        }
        function sendErr(reason, requestId = null) { send({ type: "error", reason, requestId }); }

        // Reuse the existing decodeWebSocketTextFrame pattern is private to server.mjs;
        // replicate inline here (canonical RFC 6455 single-frame text decode):
        socket.on("data", async (buf) => {
          if (buf.length < 2) return;
          const fin = (buf[0] & 0x80) !== 0;
          const opcode = buf[0] & 0x0f;
          if (opcode === 0x8) { socket.end(); return; }
          if (opcode !== 0x1) return;
          const masked = (buf[1] & 0x80) !== 0;
          let len = buf[1] & 0x7f;
          let off = 2;
          if (len === 126) { len = buf.readUInt16BE(off); off += 2; }
          else if (len === 127) { len = Number(buf.readBigUInt64BE(off)); off += 8; }
          let mask = null; if (masked) { mask = buf.subarray(off, off + 4); off += 4; }
          const payload = buf.subarray(off, off + len);
          if (masked) for (let i = 0; i < payload.length; i++) payload[i] ^= mask[i & 3];
          const text = payload.toString("utf8");
          let msg; try { msg = JSON.parse(text); } catch { sendErr("invalid-json"); return; }
          const { action, requestId } = msg;
          if (!KNOWN_ACTIONS.has(action)) { sendErr("unknown-action", requestId); return; }

          try {
            const router = getRouter();
            if (!router) { sendErr("router-not-ready", requestId); return; }

            if (action === "get-router-rtp-capabilities") {
              send({ type: "router-rtp-capabilities", requestId, rtpCapabilities: router.rtpCapabilities });
              return;
            }
            if (action === "create-send-transport" || action === "create-recv-transport") {
              const t = await createWebRtcTransport({ router });
              if (action === "create-send-transport") conn.sendTransport = t; else conn.recvTransport = t;
              send({
                type: action === "create-send-transport" ? "send-transport-created" : "recv-transport-created",
                requestId,
                id: t.id,
                iceParameters: t.iceParameters,
                iceCandidates: t.iceCandidates,
                dtlsParameters: t.dtlsParameters,
              });
              return;
            }
            if (action === "connect-transport") {
              if (!validateDtlsParameters(msg.dtlsParameters)) { sendErr("invalid-dtls", requestId); return; }
              const t = msg.transportId === conn.sendTransport?.id ? conn.sendTransport :
                        msg.transportId === conn.recvTransport?.id ? conn.recvTransport : null;
              if (!t) { sendErr("transport-not-found", requestId); return; }
              await t.connect({ dtlsParameters: msg.dtlsParameters });
              send({ type: "transport-connected", requestId });
              return;
            }
            if (action === "produce") {
              if (conn.role !== "ssr-tab") { sendErr("only-ssr-tab-can-produce", requestId); return; }
              if (!validateRtpParameters(msg.rtpParameters)) { sendErr("invalid-rtp-parameters", requestId); return; }
              if (!conn.sendTransport) { sendErr("no-send-transport", requestId); return; }
              const producer = await conn.sendTransport.produce({ kind: msg.kind, rtpParameters: msg.rtpParameters });
              conn.producers.set(producer.id, producer);
              if (msg.kind === "video") state.videoProducer = producer;
              if (msg.kind === "audio") state.audioProducer = producer;
              send({ type: "produced", requestId, id: producer.id });
              return;
            }
            if (action === "consume") {
              if (!conn.recvTransport) { sendErr("no-recv-transport", requestId); return; }
              const producer = msg.kind === "audio" ? state.audioProducer : state.videoProducer;
              if (!producer) { sendErr("no-producer-yet", requestId); return; }
              if (!router.canConsume({ producerId: producer.id, rtpCapabilities: msg.rtpCapabilities })) {
                sendErr("cannot-consume", requestId); return;
              }
              const consumer = await conn.recvTransport.consume({
                producerId: producer.id,
                rtpCapabilities: msg.rtpCapabilities,
                paused: true, // resume after client signals ready
              });
              conn.consumers.set(consumer.id, consumer);
              send({
                type: "consumed", requestId,
                id: consumer.id, producerId: producer.id,
                kind: consumer.kind, rtpParameters: consumer.rtpParameters,
              });
              return;
            }
            if (action === "resume-consumer") {
              const c = conn.consumers.get(msg.consumerId);
              if (!c) { sendErr("consumer-not-found", requestId); return; }
              await c.resume();
              send({ type: "consumer-resumed", requestId });
              return;
            }
          } catch (err) {
            logger.error(`[ssr-signal] action=${action} error: ${err?.message}`);
            sendErr(err?.message ?? "internal-error", requestId);
          }
        });

        socket.on("close", () => {
          for (const p of conn.producers.values()) try { p.close(); } catch {}
          for (const c of conn.consumers.values()) try { c.close(); } catch {}
          if (conn.sendTransport) try { conn.sendTransport.close(); } catch {}
          if (conn.recvTransport) try { conn.recvTransport.close(); } catch {}
          if (state.videoProducer && [...conn.producers.values()].includes(state.videoProducer)) state.videoProducer = null;
          if (state.audioProducer && [...conn.producers.values()].includes(state.audioProducer)) state.audioProducer = null;
          logger.info(`[ssr-signal] disconnect role=${conn.role}`);
        });
        socket.on("error", () => {}); // ignore — close handler does cleanup
      });

      return state;
    }
    ```

    ```js
        });

        });
    ```

    ```js
      const script = buildInPagePublisherScript();

      // Poll for success/failure

          return result.producers;
        }

      }

    }
    ```
