<objective>
Implement B5 — content-hash-based asset cache invalidation. Server computes
`sha256(bytes)[:12]` per upload, persists to `config/asset-manifest.json`, broadcasts
on change. Client maintains a manifest mirror and a single `ctx.resolveAssetUrlWithHash`
resolver that EVERY asset-URL consumer (gif/mp4 playback, draw loop, sound picker)
funnels through. Hash-suffixed URLs (`/resources/animations/foo.gif?v=abc123def456`)
naturally invalidate browser HTTP cache AND in-memory `Image`/`HTMLVideoElement` caches
(which are keyed by the path string).

Additionally, this plan converts the Plan 28-03 `TODO(28-04)` markers into real
hash-diff guards (B3-D07.2 + B3-D07.3) — re-uploading an asset with the same name AND
the same bytes (same hash) DOES NOT fire dirty; with different bytes (different hash)
DOES fire dirty.

rein was das verursacht und behebe es." Root cause is multi-layer (HTTP cache +
in-memory `gifPlaybackCacheByPath` Map + `outsideVideoCacheByPath` Map). Hash-suffix
URLs invalidate ALL THREE layers in one stroke because the URL string IS the cache
key.

    ```javascript
    const ASSET_MANIFEST_PATH = path.join(CONFIG_DIR, "asset-manifest.json");
    const ASSET_MANIFEST_SCHEMA = "tt-beamer.asset-manifest.v1";
    
    async function loadAssetManifest() {
      try {
        const raw = await readFile(ASSET_MANIFEST_PATH, "utf8");
        const parsed = JSON.parse(raw);
        if (parsed?.schema !== ASSET_MANIFEST_SCHEMA || !parsed.hashByPath || typeof parsed.hashByPath !== "object") {
          return { schema: ASSET_MANIFEST_SCHEMA, generatedAt: new Date().toISOString(), hashByPath: {} };
        }
        return parsed;
      } catch (error) {
        if (error?.code === "ENOENT") {
          return { schema: ASSET_MANIFEST_SCHEMA, generatedAt: new Date().toISOString(), hashByPath: {} };
        }
        console.warn("[asset-manifest] load failed:", error?.message || error);
        return { schema: ASSET_MANIFEST_SCHEMA, generatedAt: new Date().toISOString(), hashByPath: {} };
      }
    }
    
    async function saveAssetManifest(manifest) {
      manifest.schema = ASSET_MANIFEST_SCHEMA;
      manifest.generatedAt = new Date().toISOString();
      await mkdir(path.dirname(ASSET_MANIFEST_PATH), { recursive: true });
      await writeFile(ASSET_MANIFEST_PATH, JSON.stringify(manifest, null, 2) + "\\n");
    }
    
    function computeAssetHash(buffer) {
      return createHash("sha256").update(buffer).digest("hex").substring(0, 12);
    }
    
    async function synthesizeAssetManifestFromDisk() {
      const manifest = { schema: ASSET_MANIFEST_SCHEMA, generatedAt: new Date().toISOString(), hashByPath: {} };
      for (const config of [ANIMATION_RESOURCE_CONFIG, SOUND_RESOURCE_CONFIG]) {
        let files = [];
        try {
          files = await readdir(config.absoluteDir);
        } catch { continue; }
        for (const filename of files) {
          if (!config.extensions.has(filename.split(".").pop())) continue;
          const absolute = path.join(config.absoluteDir, filename);
          try {
            const buf = await readFile(absolute);
            const stat = await fsStat(absolute);
            const url = `/resources/${config.folder}/${filename}`;
            manifest.hashByPath[url] = {
              hash: computeAssetHash(buf),
              size: buf.length,
              mtime: stat.mtime.toISOString(),
            };
          } catch (err) {
            console.warn(`[asset-manifest] failed to hash ${absolute}:`, err?.message || err);
          }
        }
      }
      return manifest;
    }
    ```
